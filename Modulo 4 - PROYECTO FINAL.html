#4. Funcion para obtener puntuaciones.

def get_scores (model, xtrain, xtest, ytrain, ytest):
    
    train_pred = model.predict(xtrain)
    test_pred = model.predict(xtest)
    
    r2_train_score = r2_score(ytrain, train_pred)
    r2_test_score = r2_score(ytest, test_pred)
    
    diff_score = abs(r2_train_score - r2_test_score)
    
    mae_score = mae(ytest, test_pred)
    mse_score = mse(ytest, test_pred)
    rmae_score = np.sqrt(mae_score)
    rmse_score = np.sqrt(mse_score)
    
    test_pred2 = np.where(test_pred < 0, -test_pred, test_pred)
    
    msle_score = msle(ytest, test_pred2)
    rmsle_score = np.sqrt(msle_score)
    
    data = [r2_train_score, r2_test_score, diff_score, mae_score, mse_score, rmae_score, rmse_score, msle_score, rmsle_score]
    
    names = ['r2_train', 'r2_test', 'diff', 'mae', 'mse', 'rmae', 'rmse', 'msle', 'rmsle']
    
    scores = pd.Series(data, index = names)
    
    return scores

#5. Funcion para crear modelos.

def make_models(xtrain, xtest, ytrain, ytest):
    
    models = pd.DataFrame()
    
    model_LR = LinearRegression().fit(xtrain, ytrain)
    models['model_LR'] = round(get_scores(model_LR, xtrain, xtest, ytrain, ytest),5)
        
    for d in range(3,10):

        model_RF = RandomForestRegressor(n_estimators = 1000, max_depth = d).fit(xtrain, ytrain)
        models[f'model_RF_{d}'] = round(get_scores(model_RF, xtrain, xtest, ytrain, ytest),5)
    
    for d in range(3,10):

        model_xgb = XGBRegressor(max_depth = d, objective = 'reg:squarederror').fit(xtrain, ytrain)
        models[f'model_xgb_{d}'] = round(get_scores(model_xgb, xtrain, xtest, ytrain, ytest),5)
    
    for d in range(3,10):

        model_DT = DecisionTreeRegressor(max_depth = d).fit(xtrain, ytrain)
        models[f'model_DT_{d}'] = round(get_scores(model_DT, xtrain, xtest, ytrain, ytest),5)
        
    
    models = models.T
    
    return models

# Funcion de ajuste y contraste 

def make_models_Poly(xtrain, xtest, ytrain, ytest):
    
    models = pd.DataFrame()
    
    for a in [0.001, 0.05, 0.01, 0.5, 0.1, 1]:

        model_lasso = Lasso(alpha = a, tol = 0.1).fit(xtrain, ytrain)
        models[f'model_lasso_{a}'] = round(get_scores(model_lasso, xtrain, xtest, ytrain, ytest),5)

    for a in [0.001, 0.05, 0.01, 0.5, 0.1, 1]:

        model_ridge = Ridge(alpha = a, tol = 0.1).fit(xtrain, ytrain)
        models[f'model_ridge_{a}'] = round(get_scores(model_ridge, xtrain, xtest, ytrain, ytest),5)

    
    models = models.T
    
    return models
